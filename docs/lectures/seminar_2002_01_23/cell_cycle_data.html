Example: S. cerevisiae cell cycle

A possible data format for the description of a biological process: 



<?xml version="1.0"?>
<biological_process title="S. cerevisiae cell cycle">

  <duration units="min">
    <value>110</value><stderr>20</stderr>
  </duration>

  <phases>
    <phase name="G1">
      <duration units="min">
        <value>30</value><stderr>8</stderr>
      </duration>
    </phase>
    ...
  </phases>

  <property>
    <expression>
      <gene>CLN1</gene>
      <point>
        <time type="absolute" phase="G1">0.4</time>
        <value type="relative">1.2</value>
      </point>
      ...
  </property>
</biological_process>
